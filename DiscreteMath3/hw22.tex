\section*{Homework 22}

\paragraph{Homework 22.}
Let $p \in (0, 1)$. Consider a sequence of $n$ independent coin flips in which the coin will land on heads
with probability $p$ and tails with probability $1 - p$. Define $X$ to be the random variable that represents
the number of heads in the sequence. Show that for any $\varepsilon > 0$,
\begin{equation}\label{eq:upper-limit-cheb}
    \lim_{n \to \infty} \Pr\!\bigl(X \ge (1 + \varepsilon)\mathbb{E}(X)\bigr) = 0
\end{equation}
and
\begin{equation}\label{eq:lower-limit-cheb}
    \lim_{n \to \infty} \Pr\!\bigl(X \le (1 - \varepsilon)\mathbb{E}(X)\bigr) = 0.
\end{equation}
\medskip

\begin{proof}[解答]
    $X$ は $n$ 回のコイン投げにおける表の回数を表すので,$X$は二項分布$B(n,p)$に従う.
    二項分布の基本的な性質より
    \begin{equation}\label{eq:EX-varX}
        \mathbb{E}(X) = np,\qquad
        \operatorname{Var}(X) = np(1-p)
    \end{equation}
    が成り立つ．

    チェビシェフ不等式より，任意の $a>0$ について
    \begin{equation}\label{eq:chebyshev}
        \Pr\bigl(|X - \mathbb{E}(X)| \ge a\bigr)
        \le \frac{\operatorname{Var}(X)}{a^2}
    \end{equation}
    が成り立つ．ここで
    \[
        a := \varepsilon \mathbb{E}(X)
    \]
    とおくと，\eqref{eq:chebyshev} と \eqref{eq:EX-varX} より
    \begin{equation}\label{eq:cheb-with-a}
        \Pr\bigl(|X - \mathbb{E}(X)| \ge \varepsilon \mathbb{E}(X)\bigr)
        \le
        \frac{\operatorname{Var}(X)}{\varepsilon^2 \mathbb{E}(X)^2}
        = \frac{np(1-p)}{\varepsilon^2 n^2 p^2}
        = \frac{1-p}{\varepsilon^2 n p}.
    \end{equation}

    \medskip
    \noindent\textbf{式 \eqref{eq:upper-limit-cheb} の証明.}
    任意の $\varepsilon>0$ について
    \[
        X \ge (1+\varepsilon)\mathbb{E}(X)
        \;\Rightarrow\;
        X - \mathbb{E}(X) \ge \varepsilon \mathbb{E}(X)
        \;\Rightarrow\;
        |X - \mathbb{E}(X)| \ge \varepsilon \mathbb{E}(X)
    \]
    である．したがって
    \[
        \Pr\!\bigl(X \ge (1+\varepsilon)\mathbb{E}(X)\bigr)
        \le
        \Pr\bigl(|X - \mathbb{E}(X)| \ge \varepsilon \mathbb{E}(X)\bigr).
    \]
    これと \eqref{eq:cheb-with-a} より
    \[
        0 \le
        \Pr\!\bigl(X \ge (1+\varepsilon)\mathbb{E}(X)\bigr)
        \le \frac{1-p}{\varepsilon^2 n p}
    \]
    となる．右辺は $n \to \infty$ で $0$ に収束するので，はさみうちの原理から
    \[
        \lim_{n \to \infty}
        \Pr\!\bigl(X \ge (1+\varepsilon)\mathbb{E}(X)\bigr)
        = 0,
    \]
    すなわち \eqref{eq:upper-limit-cheb} が従う．

    \medskip
    \noindent\textbf{式 \eqref{eq:lower-limit-cheb} の証明.}
    任意の $\varepsilon>0$ について
    \[
        X \le (1-\varepsilon)\mathbb{E}(X)
        \;\Rightarrow\;
        \mathbb{E}(X) - X \ge \varepsilon \mathbb{E}(X)
        \;\Rightarrow\;
        |X - \mathbb{E}(X)| \ge \varepsilon \mathbb{E}(X)
    \]
    である．したがって
    \[
        \Pr\!\bigl(X \le (1-\varepsilon)\mathbb{E}(X)\bigr)
        \le
        \Pr\bigl(|X - \mathbb{E}(X)| \ge \varepsilon \mathbb{E}(X)\bigr).
    \]
    \eqref{eq:cheb-with-a} を用いると
    \[
        0 \le
        \Pr\!\bigl(X \le (1-\varepsilon)\mathbb{E}(X)\bigr)
        \le \frac{1-p}{\varepsilon^2 n p}
    \]
    となり，右辺は $n \to \infty$ で $0$ に収束するので，はさみうちの原理より
    \[
        \lim_{n \to \infty}
        \Pr\!\bigl(X \le (1-\varepsilon)\mathbb{E}(X)\bigr)
        = 0,
    \]
    すなわち \eqref{eq:lower-limit-cheb} が成り立つ．

    以上より，任意の $\varepsilon > 0$ について
    \eqref{eq:upper-limit-cheb} と \eqref{eq:lower-limit-cheb} が示された．
\end{proof}
