\section*{Homework 18}

\paragraph{Homework 18.}
Let $k \geq 1$ be a positive integer and $X$ a discrete random variable with a finite $k$th moment. Show that for any nonnegative integer $l \leq k$, the $l$th moment $\mathbb{E}\!\left(X^l\right)$ is finite.
(Hint: if $0 \leq l \leq k$, then for any $i \in \mathbb{R}$, $\vert i\vert^l \leq \max\!\left(1, \vert i\vert^k\right)$.)
\medskip

\begin{proof}[解答]
    $0 \le l \le k$ をみたす整数 $l$ を取る．任意の $i \in \mathbb{R}$ について
    \[
        |i|^l \le \max(1, |i|^k) \le 1 + |i|^k
    \]
    が成り立つ．したがって
    \[
        \mathbb{E}\!\left(|X|^l\right)=\sum_{i} |i|^l \Pr(X = i)
        \le \sum_{i} (1 + |i|^k)\Pr(X = i)
        = 1 + \mathbb{E}\!\left(|X|^k\right).
    \]
    仮定より $\mathbb{E}\!\left(|X|^k\right)$ が有限なので，$1+\mathbb{E}\!\left(|X|^k\right)$ は有限である．
    よって$\mathbb{E}\!\left(|X|^l\right)$ も有限である．

    また $-|X|^l \le X^l \le |X|^l$ より
    \[
        -\mathbb{E}\!\left(|X|^l\right)
        \le \mathbb{E}\!\left(X^l\right)
        \le \mathbb{E}\!\left(|X|^l\right)
    \]
    が成り立つから，$\mathbb{E}\!\left(X^l\right)$ も有限である．
\end{proof}
